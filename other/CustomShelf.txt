def shelf(name="GraphNTools"):

    # get top shelf
    gShelfTopLevel = mel.eval("$tmpVar=$gShelfTopLevel")

    # get top shelf names
    shelves = cmds.tabLayout(gShelfTopLevel, query=1, ca=1)
    
    # delete shelf if it exists
    if name in shelves:
        print(">> Removing old GraphNTools shelf before creating a new one!")
        cmds.deleteUI(name)

    # create shelf
    cmds.shelfLayout(name, parent=gShelfTopLevel)
    
    # let's add some test button on it
    cmds.shelfButton(image1=os.path.join(DIR_, "res", "file.svg"), command=shelf_toolLib_cmd, parent=name)
